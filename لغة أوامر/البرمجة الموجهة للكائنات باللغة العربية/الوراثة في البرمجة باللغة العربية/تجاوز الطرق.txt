أكتب مباشرة الكود البرمجي المقترح بشكل صحيح وجملة صحيحة يمكن قرائتها وفهمها من قبل الجميع باللغة العربية الفصحى مع ذكر مثال لطريقة تطبيق هذا الأمر البرمجي 

```برمجة
//هذا الكود هو مثال على تجاوز الطرق في الوراثة باللغة العربية

/// الصف الأساسي هو "طالب"
صف طالب {
    نص اسم;
    رقم عمر;
    
    // هنا نحدد الدالة الأساسية للطالب
    دالة قارئة نص قراءة_معلومات () {
        ارجع "اسم الطالب هو " + اسم + " وعمره " + عمر + " سنة.";
    }
}

// الصف المشتق هو "طالب_جامعي"
صف طالب_جامعي: عمومي طالب {
    نص اختصاص;
    
    // هنا نقوم بتجاوز الدالة "قراءة_معلومات"
    دالة قارئة نص قراءة_معلومات () {
        ارجع "اسم الطالب هو " + اسم + " وعمره " + عمر + " سنة. يدرس " + اختصاص;
    }
}

 رئيسي
{
طالب الطالب1 = جديد طالب("محمد", 21);
طالب_جامعي الطالب2  = جديد طالب_جامعي("علي", 20, "هندسة الحاسوب");

طباعة(الطالب1.قراءة_معلومات());
طباعة(الطالب2.قراءة_معلومات());

/* النتيجة:
    اسم الطالب هو محمد وعمره 21 سنة.,
    اسم الطالب هو علي وعمره 20 سنة. يدرس هندسة الحاسوب
*/
}
```
