أكتب مباشرة الكود البرمجي المقترح بشكل صحيح وجملة صحيحة يمكن قرائتها وفهمها من قبل الجميع باللغة العربية الفصحى مع ذكر مثال لطريقة تطبيق هذا الأمر البرمجي 

لا يوجد لغة برمجة باللغة العربية أو باللغة العربية الفصحى حاليا، ولكن أعرض لك حالة وهمية عن كيفية كتابة الكود البرمجي باللغة العربية الفصحى لتعريف الكائنات والصفوف. 

```شفرة
صنف إنسان {
    خصائص {
        اسم: خيط,
        سن: رقم,
    }

    الدوال {
        نشاط_التحية() {
            اطبع("السلام عليكم، أنا " + هذا.اسم)
        }
    }
}

كائن أحمد من صنف إنسان {
    اسم: "أحمد",
    سن: 35,
}

أحمد.نشاط_التحية()
```
وسوف أقدم لكم مثالا آخرا حول كيفية تعريف كائن مع وظائف محددة بشكل صحيح باللغة العربية الفصحى:

```شفرة
صنف سيارة {
    خصائص {
        نوع: خيط,
        سرعة: رقم,
    }

    الدوال {
        زيادة_السرعة(مقدار) {
            هذا.سرعة= هذا.سرعة+ مقدار;
        }
        أظهر_السرعة() {
            اطبع ("السرعة الحالية : "+ هذا.سرعة)
        }
    }
}

كائن أوبل من صنف سيارة {
    نوع: "أوبل",
    سرعة: 0,
}

أوبل.زيادة_السرعة(10)
أوبل.أظهر_السرعة()
```
احتدون على نفس الخطوات المذكورة في الأمثلة، يمكنك تعريف أي عدد تريده من الكائنات والتعامل معها بسهولة.
